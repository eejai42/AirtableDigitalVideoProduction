The TypeScript libraries have been tested with Angular2 and Ionic2.

Configuration

You'll need to add the the followin libraries, as well as adding references to each actor manually in the app.modules.ts.

  1. $ npm install @stomp/ng2-stompjs --save
  2. $ npm install angular2-uuid --save

Programmer


import { Component } from '@angular/core';
...
import { SMQProgrammer } from '../../../providers/sassymq/gss/smqProgrammer';
...
export class Home {

    constructor(public navCtrl: NavController, 
                public navParams: NavParams, 
                public programmer: SMQProgrammer) {

         // Try "Saying" some things
        
        
         // Say "Hello" to World
         // 
         var payload = this.programmer.createPayload();
         payload. = {...};
         this.programmer.Hello(payload, (reply) {
            // Do something here with reply.body.
         });
        
        
         // Say "Goodbye" to World
         // 
         var payload = this.programmer.createPayload();
         payload. = {...};
         this.programmer.Goodbye(payload, (reply) {
            // Do something here with reply.body.
         });
        

        
         // Listen for messages
        
        
         // Hear Wassup from World
         // 
         this.programmer.onWassup((message) {
            // Do something here with message.body.
         });
        

    }

}
